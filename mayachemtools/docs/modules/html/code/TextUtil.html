  40 # Add number suffix...
  41 sub AddNumberSuffix {
  42   my($Value) = @_;
  43   my($ValueWithSuffix, $Suffix);
  44 
  45   $ValueWithSuffix = $Value;
  46   if (!IsPositiveInteger($Value)) {
  47     return $ValueWithSuffix;
  48   }
  49   $Suffix = "th";
  50   if ($Value < 10 || $Value > 20) {
  51     my $Remainder = $Value % 10;
  52     $Suffix = ($Remainder == 1) ? "st" : (($Remainder == 2) ? "nd" : (($Remainder == 3) ? "rd" : "th"));
  53   }
  54   $ValueWithSuffix = "${ValueWithSuffix}${Suffix}";
  55   return $ValueWithSuffix;
  56 }

  69 # Read the line, change to UNIX new line char, and chop off new line char as well...
  70 sub GetTextLine {
  71   my($TextFileRef) = @_;
  72   my($Line) = '';
  73 
  74   # Get the next non empty line...
  75   LINE: while (defined($_ = <$TextFileRef>)) {
  76     # Change Windows and Mac new line char to UNIX...
  77     s/(\r\n)|(\r)/\n/g;
  78 
  79     # Take out any new line char at the end by explicitly removing it instead of using
  80     # chomp, which might not always work correctly on files generated on a system
  81     # with a value of input line separator different from the current system...
  82     s/\n$//g;
  83 
  84     # Doesn't hurt to chomp...
  85     chomp;
  86 
  87     $Line = $_;
  88     if (length $Line) {
  89       last LINE;
  90     }
  91   }
  92   return $Line;
  93 }

  95 # Load data from a CSV file into the specified hash reference using a specific
  96 # column for unique data key values.
  97 #
  98 # The lines starting with # are treated as comments and ignored. First line
  99 # not starting with # must contain column labels and the number of columns in
 100 # all other data rows must match the number of column labels.
 101 #
 102 # The first column is assumed to contain data key value by default; all other columns
 103 # contain data as indicated in their column labels.
 104 #
 105 # In order to avoid dependence of data access on the specified column labels, the
 106 # column data is loaded into hash with Column<Num> hash keys, where column number
 107 # start from 1. The data key column is not available as Colnum<Num> hash key;
 108 #
 109 # The format of the data structure loaded into a specified hash reference is:
 110 #
 111 # @{$TextDataMapRef->{DataKeys}} - Array of unique data keys
 112 # @{$TextDataMapRef->{ColLabels}} - Array of column labels
 113 # @{$TextDataMapRef->{DataColIDs}} - Array of data column IDs
 114 # $TextDataMapRef->{NumOfCols} - Number of columns
 115 # %{$TextDataMapRef->{DataKey}} - Hash keys pair: <DataKey, DataKey>
 116 # %{$TextDataMapRef->{DataCol<Num>}} - Hash keys pair: <DataCol<Num>, DataKey>
 117 #
 118 # Caveats:
 119 #   . The column number start from 1.
 120 #   . Column data for data key column column is not loaded into <Column<Num>, DataKey> hash keys pairs.
 121 #
 122 sub GetTextFileDataByUniqueKey {
 123   my($TextDataFile, $TextDataMapRef, $DataKeyColNum, $InDelim) = @_;
 124 
 125   return _GetTextFileData("UniqueKey", $TextDataFile, $TextDataMapRef, $DataKeyColNum, $InDelim);
 126 }

 161 # Loadtext file data using unique or non-uniqye data column key...
 162 #
 163 sub _GetTextFileData {
 164   my($DataKeyMode, $TextDataFile, $TextDataMapRef, $DataKeyColNum, $InDelim) = @_;
 165   my($DataKeyColIndex, $LineCount, $IgnoredLineCount, $UniqueDataKeyMode, $DataKey, $Line, $NumOfCols, $ColIndex, $ColNum, $ColID, $ColValue, @LineWords, @ColLabels, @DataColIDs, @DataColNums);
 166 
 167   print "\nProcessing text data file $TextDataFile...\n";
 168 
 169   $UniqueDataKeyMode = 0;
 170   if ($DataKeyMode =~ /^UniqueKey$/i) {
 171     $UniqueDataKeyMode = 1;
 172   }
 173 
 174   # Setup default values...
 175   $DataKeyColNum = defined  $DataKeyColNum ? $DataKeyColNum : 1;
 176 
 177   if (defined $InDelim) {
 178     if ($InDelim =~ /^tab$/i) {
 179       $InDelim = "\t";
 180     }
 181     elsif ($InDelim =~ /^semicolon$/i) {
 182       $InDelim = "\;";
 183     }
 184     elsif ($InDelim =~ /^comma$/i) {
 185       $InDelim = "\,";
 186     }
 187     else {
 188       warn "Warning: Ignoring specified input delimiter: $InDelim. Supported values: comma, semicolon or tab. Using default comma delimiter...";
 189       $InDelim = "\,";
 190     }
 191   }
 192   else {
 193     if ($TextDataFile =~ /\.tsv$/i) {
 194       $InDelim = "\t";
 195     }
 196     elsif ($TextDataFile =~ /\.csv$/i) {
 197       $InDelim = "\,";
 198     }
 199     else {
 200       warn "Warning: Unknown file extension. Using default comma delimiter...";
 201       $InDelim = "\,";
 202     }
 203   }
 204 
 205   ($LineCount, $IgnoredLineCount) = (0) x 2;
 206 
 207   open TEXTDATAFILE, "$TextDataFile" or die "Couldn't open $TextDataFile: $! ...";
 208 
 209   # Skip lines up to column labels...
 210   LINE: while ($Line = TextUtil::GetTextLine(\*TEXTDATAFILE)) {
 211     $LineCount++;
 212     if ($Line =~ /^#/) {
 213       $IgnoredLineCount++;
 214     }
 215     else {
 216       last LINE;
 217     }
 218   }
 219 
 220   # Initialize data map...
 221   %{$TextDataMapRef} = ();
 222   @{$TextDataMapRef->{DataKeys}} = ();
 223   @{$TextDataMapRef->{ColLabels}} = ();
 224   @{$TextDataMapRef->{DataColIDs}} = ();
 225   $TextDataMapRef->{NumOfCols} = undef;
 226 
 227   # Process column labels...
 228   @ColLabels= quotewords($InDelim, 0, $Line);
 229   $NumOfCols = @ColLabels;
 230 
 231   if ($DataKeyColNum < 1 || $DataKeyColNum > $NumOfCols) {
 232     warn "Warning: Ignoring text data file $TextDataFile: Invalid data key column number, $DataKeyColNum, specified. It must be > 0 or <= $NumOfCols, number of columns in the text file ...";
 233     return;
 234   }
 235   $DataKeyColIndex = $DataKeyColNum - 1;
 236 
 237   $TextDataMapRef->{NumOfCols} = $NumOfCols;
 238   push @{$TextDataMapRef->{ColLabels}}, @ColLabels;
 239 
 240   # Set up column data IDs for tracking the data...
 241   @DataColNums = ();
 242   @DataColIDs = ();
 243   COLNUM: for $ColNum (1 .. $NumOfCols) {
 244     if ($ColNum == $DataKeyColNum) {
 245       next COLNUM;
 246     }
 247     push @DataColNums, $ColNum;
 248     $ColID = "DataCol${ColNum}";
 249     push @DataColIDs, $ColID;
 250   }
 251   push @{$TextDataMapRef->{DataColIDs}}, @DataColIDs;
 252 
 253   # Initialize column data hash...
 254   %{$TextDataMapRef->{DataKey}} = ();
 255   for $ColIndex (0 .. $#DataColNums) {
 256     $ColNum = $DataColNums[$ColIndex];
 257     $ColID = $DataColIDs[$ColIndex];
 258     %{$TextDataMapRef->{$ColID}} = ();
 259   }
 260 
 261   LINE: while ($Line = TextUtil::GetTextLine(\*TEXTDATAFILE)) {
 262     $LineCount++;
 263     if ($Line =~ /^#/) {
 264       $IgnoredLineCount++;
 265       next LINE;
 266     }
 267 
 268     @LineWords = quotewords($InDelim, 0, $Line);
 269     if (@LineWords != $NumOfCols) {
 270       $IgnoredLineCount++;
 271       warn "Warning: The number of data fields, @LineWords, in $TextDataFile must be $NumOfCols.\nIgnoring line number $LineCount: $Line...\n";
 272       next LINE;
 273     }
 274     $DataKey = $LineWords[$DataKeyColIndex];
 275 
 276     if ($UniqueDataKeyMode) {
 277       if (exists $TextDataMapRef->{DataKey}{$DataKey}) {
 278         $IgnoredLineCount++;
 279         warn "Warning: The data key, $DataKey, in data column key number, $DataKeyColNum, is already present.\nIgnoring line number $LineCount: $Line...\n";
 280         next LINE;
 281       }
 282       push @{$TextDataMapRef->{DataKeys}}, $DataKey;
 283       $TextDataMapRef->{DataKey}{$DataKey} = $DataKey;
 284     }
 285     else {
 286       if (!exists $TextDataMapRef->{DataKey}{$DataKey}) {
 287         push @{$TextDataMapRef->{DataKeys}}, $DataKey;
 288         $TextDataMapRef->{DataKey}{$DataKey} = $DataKey;
 289 
 290         for $ColIndex (0 .. $#DataColNums) {
 291           $ColNum = $DataColNums[$ColIndex];
 292           $ColID = $DataColIDs[$ColIndex];
 293           @{$TextDataMapRef->{$ColID}{$DataKey}} = ();
 294         }
 295       }
 296     }
 297 
 298     # Track column data values...
 299     for $ColIndex (0 .. $#DataColNums) {
 300       $ColID = $DataColIDs[$ColIndex];
 301 
 302       $ColNum = $DataColNums[$ColIndex];
 303       $ColValue = $LineWords[$ColNum - 1];
 304 
 305       if ($UniqueDataKeyMode) {
 306         $TextDataMapRef->{$ColID}{$DataKey} = $ColValue;
 307       }
 308       else {
 309         push @{$TextDataMapRef->{$ColID}{$DataKey}}, $ColValue;
 310       }
 311     }
 312 
 313   }
 314 
 315   print "\nTotal number of lines in file $TextDataFile: $LineCount\n";
 316   print "Total number of lines ignored: $IgnoredLineCount\n";
 317 
 318   close TEXTDATAFILE;
 319 }

 321 # Returns a 32 bit integer hash code using One-at-a-time algorithm By Bob Jenkins [Ref 38]. It's also implemented in
 322 # Perl for internal hash keys in hv.h include file.
 323 #
 324 # It's not clear how to force Perl perform unsigned integer arithmetic irrespective of the OS/Platform and
 325 # the value of use64bitint flag used during its compilation.
 326 #
 327 # In order to generate a consistent 32 bit has code across OS/platforms, the following methodology appear
 328 # to work:
 329 #
 330 #    o Use MaxHashCodeMask to retrieve appropriate bits after left shifting by bit operators and additions
 331 #    o Stay away from "use integer" to avoid signed integer arithmetic for bit operators
 332 #
 333 #
 334 #   MaxHashCodeMask (2147483647) corresponds to the maximum value which can be stored in 31 bits
 335 #
 336 my($MaxHashCodeMask);
 337 $MaxHashCodeMask = 2**31 - 1;
 338 
 339 sub HashCode {
 340   my($String) = @_;
 341   my($HashCode, $Value, $ShiftedHashCode);
 342 
 343   $HashCode = 0;
 344   for $Value (unpack('C*', $String)) {
 345     $HashCode += $Value;
 346 
 347     $ShiftedHashCode = $HashCode << 10;
 348     if ($ShiftedHashCode > $MaxHashCodeMask) {
 349       $ShiftedHashCode = $ShiftedHashCode & $MaxHashCodeMask;
 350     }
 351 
 352     $HashCode += $ShiftedHashCode;
 353     if ($HashCode > $MaxHashCodeMask) {
 354       $HashCode = $HashCode & $MaxHashCodeMask;
 355     }
 356 
 357     $HashCode ^= ($HashCode >> 6);
 358   }
 359 
 360   $ShiftedHashCode = $HashCode << 3;
 361   if ($ShiftedHashCode > $MaxHashCodeMask) {
 362     $ShiftedHashCode = $ShiftedHashCode & $MaxHashCodeMask;
 363   }
 364 
 365   $HashCode += $ShiftedHashCode;
 366   if ($HashCode > $MaxHashCodeMask) {
 367     $HashCode = $HashCode & $MaxHashCodeMask;
 368   }
 369   $HashCode ^= ($HashCode >> 11);
 370 
 371   $ShiftedHashCode = $HashCode << 15;
 372   if ($ShiftedHashCode > $MaxHashCodeMask) {
 373     $ShiftedHashCode = $ShiftedHashCode & $MaxHashCodeMask;
 374   }
 375 
 376   $HashCode += $ShiftedHashCode;
 377   if ($HashCode > $MaxHashCodeMask) {
 378     $HashCode = $HashCode & $MaxHashCodeMask;
 379   }
 380   return $HashCode;
 381 }

 393 # Is first specified number power of second specified number...
 394 sub IsNumberPowerOfNumber {
 395   my($FirstNum, $SecondNum) = @_;
 396   my($PowerValue);
 397 
 398   $PowerValue = log($FirstNum)/log($SecondNum);
 399 
 400   return IsInteger($PowerValue) ? 1 : 0;

 403 # Check out the string: Is it an integer?
 404 sub IsInteger {
 405   my($TheString) = @_;
 406   my($Status) = 0;
 407 
 408   if (defined($TheString) && length($TheString)) {
 409     $TheString = RemoveLeadingAndTrailingWhiteSpaces($TheString);
 410     $TheString =~ s/^[+-]//;
 411     $Status = ($TheString =~ /[^0-9]/) ? 0 : 1;
 412   }
 413   return $Status;
 414 }

 463 # Join different words using delimiter and quote parameters. And return as
 464 # a string value.
 465 sub JoinWords {
 466   my($Words, $Delim, $Quote) = @_;
 467 
 468   if (!@$Words) {
 469     return "";
 470   }
 471 
 472   $Quote = $Quote ? "\"" : "";
 473   my(@NewWords) = map { (defined($_) && length($_)) ? "${Quote}$_${Quote}" : "${Quote}${Quote}" } @$Words;
 474 
 475   return join $Delim, @NewWords;
 476 }

 478 # Split string value containing quoted or unquoted words in to an array containing
 479 # unquoted words.
 480 #
 481 # This function is used to split strings generated by JoinWords.
 482 #
 483 sub SplitWords {
 484   my($Line, $Delim) = @_;
 485 
 486   if (!$Line) {
 487     return ();
 488   }
 489 
 490   # Is it a quoted string?
 491   if ($Line =~ /^\"/) {
 492     # Take out first and last quote...
 493     $Line =~ s/^\"//; $Line =~ s/\"$//;
 494 
 495     $Delim = "\"$Delim\"";
 496   }
 497   return split /$Delim/, $Line;
 498 }

 500 # Based on quote parameter, figure out what to do
 501 sub QuoteAWord {
 502   my($Word, $Quote) = @_;
 503   my($QuotedWord);
 504 
 505   $QuotedWord = "";
 506   if ($Word) {
 507     $QuotedWord = $Word;
 508   }
 509   if ($Quote) {
 510     $QuotedWord = "\"$QuotedWord\"";
 511   }
 512   return ($QuotedWord);

 515 # Remove leading white space characters from the string...
 516 sub RemoveLeadingWhiteSpaces {
 517   my($InString) = @_;
 518   my($OutString, $TrailingString, $LeadingWhiteSpace);
 519 
 520   $OutString = $InString;
 521   if (length($InString) && ContainsWhiteSpaces($InString)) {
 522     $OutString =~ s/^([ \t\r\n\f]*)(.*?)$/$2/;
 523   }
 524   return $OutString;
 525 }

 527 # Remove Trailing white space characters from the string...
 528 sub RemoveTrailingWhiteSpaces {
 529   my($InString) = @_;
 530   my($OutString, $LeadingString, $TrailingWhiteSpace);
 531 
 532   $OutString = $InString;
 533   if (length($InString) && ContainsWhiteSpaces($InString)) {
 534     $OutString =~ s/^(.*?)([ \t\r\n\f]*)$/$1/;
 535   }
 536   return $OutString;
 537 }

 539 # Remove both leading and trailing white space characters from the string...
 540 sub RemoveLeadingAndTrailingWhiteSpaces {
 541   my($InString) = @_;
 542   my($OutString);
 543 
 544   $OutString = $InString;

 547   }
 548   return $OutString;

 551 # Wrap text string...
 552 sub WrapText {
 553   my($InString, $WrapLength, $WrapDelimiter);
 554   my($OutString);
 555 
 556   $WrapLength = 40;
 557   $WrapDelimiter = "\n";
 558   if (@_ == 3) {
 559     ($InString, $WrapLength, $WrapDelimiter) = @_;
 560   }
 561   elsif (@_ == 2) {
 562     ($InString, $WrapLength) = @_;
 563   }
 564   else {
 565     ($InString, $WrapLength) = @_;
 566   }
 567   $OutString = $InString;
 568   if ($InString && (length($InString) > $WrapLength)) {
 569     $OutString = "";
 570     my($Index, $Length, $FirstPiece, $StringPiece);
 571     $Index = 0; $Length = length($InString);
 572     $FirstPiece = 1;
 573     for ($Index = 0; $Index < $Length; $Index += $WrapLength) {
 574       if (($Index + $WrapLength) < $Length) {
 575         $StringPiece = substr($InString, $Index, $WrapLength);
 576       }
 577       else {
 578         # Last piece of the string...
 579         $StringPiece = substr($InString, $Index, $WrapLength);
 580       }
 581       if ($FirstPiece) {
 582         $FirstPiece = 0;
 583         $OutString = $StringPiece;
 584       }
 585       else {
 586         $OutString .= "${WrapDelimiter}${StringPiece}";
 587       }
 588     }
 589   }
 590   return $OutString;
 591 }
